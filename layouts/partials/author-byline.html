{{- /* Renders the author byline for a page whose `.Params.author` is a TinaCMS reference path like `content/authors/joe-taylor.md`. Params: page: Page context (defaults to dot). variant: "full" (default) | "compact" | "card". meta: Optional HTML string rendered below the name (e.g. formatted date). */ -}} {{- $page := .page | default . -}} {{- $variant := .variant | default "full" -}} {{- $meta := .meta | default "" -}} {{- $authorRef := $page.Params.author -}} {{- $author := "" -}} {{- with $authorRef -}} {{- $lookup := . | strings.TrimPrefix "content/" | strings.TrimSuffix ".md" -}} {{- $author = site.GetPage $lookup -}} {{- end -}} {{- if $author -}} {{- $name := $author.Params.name | default $author.Title -}} {{- $img := $author.Params.profile_image -}} {{- $link := $author.Params.link -}} {{- $wrapperClass := printf "author-byline author-byline--%s" $variant -}} {{- with $img -}} {{- if $link -}} {{- end -}} {{- if $link }}{{ end -}} {{- end }} {{- if $link -}} {{ $name }} {{- else -}} {{ $name }} {{- end -}} {{- with $meta }}{{ . | safeHTML }}{{ end -}} {{- else if $authorRef -}} {{- /* Fallback: front matter still holds a plain string name. */ -}} {{ $authorRef }} {{- with $meta }}{{ . | safeHTML }}{{ end -}} {{- end -}}
